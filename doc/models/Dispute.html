def self.with_role_condition(admin)
  scope = all

  case admin.role.name.to_sym
  when :system_admin, :super_admin
    scope
  when :parking_admin
    admin.disputes
  when :town_manager
    admin.disputes.joins(admin: :role).where(disputes: { admins: { roles: { name: :town_manager } } })
  else
    scope.none
  end
end
